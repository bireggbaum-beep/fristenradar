1) /opt/fristenradar/app/google_calendar.py

Inhalt/Logik:
- liest OAuth-Credentials aus /opt/fristenradar/token.json
- refresht Token bei Bedarf
- baut Google Calendar API Service
- holt Events aus Kalender-ID aus .env, default "primary"
- normalisiert Events auf:

    "id": ...,
    "title": ...,
    "note": ...,
    "start": ...,
    "end": ...,
    "allDay": ...,
    "status": ...
  }

Sinngemäßer Inhalt:

import os
from datetime import datetime, timedelta, timezone

from google.oauth2.credentials import Credentials
from google.auth.transport.requests import Request
from googleapiclient.discovery import build

SCOPES = ["https://www.googleapis.com/auth/calendar.readonly"]


def get_google_service():
    creds = Credentials.from_authorized_user_file("/opt/fristenradar/token.json", SCOPES)

    if creds.expired and creds.refresh_token:
        creds.refresh(Request())
        with open("/opt/fristenradar/token.json", "w", encoding="utf-8") as f:
            f.write(creds.to_json())

    return build("calendar", "v3", credentials=creds)


def normalize_event(event: dict):
    start = event.get("start", {})
    end = event.get("end", {})
    all_day = "date" in start

    return {
        "id": event.get("id"),
        "title": event.get("summary") or "(ohne Titel)",
        "note": event.get("description") or "",
        "start": start.get("date") if all_day else start.get("dateTime"),
        "end": end.get("date") if all_day else end.get("dateTime"),
        "allDay": all_day,
        "status": event.get("status"),
    }


def get_upcoming_events(days: int = 30):
    service = get_google_service()
    calendar_id = os.getenv("GOOGLE_CALENDAR_ID", "primary")

    now = datetime.now(timezone.utc)
    time_max = now + timedelta(days=days)

    result = (
        service.events()
        .list(
            calendarId=calendar_id,
            timeMin=now.isoformat(),
            timeMax=time_max.isoformat(),
            singleEvents=True,
            orderBy="startTime",
        )
        .execute()
    )

    events = result.get("items", [])
    return [normalize_event(event) for event in events if event.get("status") != "cancelled"]

4) /opt/fristenradar/scripts/google_auth.py

WICHTIG:
- Der ursprüngliche Browser-/Tunnel-Weg war problematisch.
- Die funktionierende Lösung war ein manuelles OAuth-Skript:
  - Script gibt Auth-URL aus
  - URL im Browser am PC öffnen
  - nach Zustimmung Redirect-URL aus Browser kopieren
  - komplette Redirect-URL im Server-Terminal einfügen
  - token.json wird geschrieben
- Zusätzlich musste gesetzt werden:
  OAUTHLIB_INSECURE_TRANSPORT=1
  weil localhost/127.0.0.1 Redirect via HTTP genutzt wurde

Sinngemäßer finaler Inhalt:

import os
os.environ["OAUTHLIB_INSECURE_TRANSPORT"] = "1"

from google_auth_oauthlib.flow import Flow

SCOPES = ["https://www.googleapis.com/auth/calendar.readonly"]
REDIRECT_URI = "http://127.0.0.1:8765/"

flow = Flow.from_client_secrets_file(
    "/opt/fristenradar/credentials.json",
    scopes=SCOPES,
    redirect_uri=REDIRECT_URI,
)

auth_url, _ = flow.authorization_url(
    access_type="offline",
    prompt="consent",
)

print(auth_url)
redirect_response = input("Komplette Redirect-URL hier einfügen: ").strip()

flow.fetch_token(authorization_response=redirect_response)

with open("/opt/fristenradar/token.json", "w", encoding="utf-8") as f:
    f.write(flow.credentials.to_json())

print("token.json geschrieben")
